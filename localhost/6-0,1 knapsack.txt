#include <bits/stdc++.h>
using namespace std;

struct Item {
    string name;
    int weight;
    int value;
    int priority;
};

bool compare(Item a, Item b) {
    if (a.priority == b.priority)
        return a.value > b.value;
    return a.priority < b.priority;
}

int main() {
    int W;
    cout << "Enter truck capacity (kg): ";
    cin >> W;

    vector<Item> items = {
        {"Medical Kits", 10, 100, 1},
        {"Food Packets", 20, 60, 2},
        {"Drinking Water", 30, 80, 2},
        {"Blankets", 15, 40, 3},
        {"Infant Formula", 5, 50, 1}
    };
    int N = items.size();
    sort(items.begin(), items.end(), compare);
    vector<vector<int>> dp(N+1, vector<int>(W+1, 0));
    for (int i = 1; i <= N; i++) {
        for (int w = 0; w <= W; w++) {
            if (items[i-1].weight <= w)
                dp[i][w] = max(dp[i-1][w],dp[i-1][w - items[i-1].weight] + items[i-1].value);
            else
                dp[i][w] = dp[i-1][w];
        }
    }

    int w = W;
    cout << "\nItems selected:\n";
    for (int i = N; i > 0 && w > 0; i--) {
        if (dp[i][w] != dp[i-1][w]) {
            cout << " - " << items[i-1].name << " (Weight: " << items[i-1].weight << ", Value: " << items[i-1].value << ", Priority: " << items[i-1].priority << ")\n";
            w -= items[i-1].weight;
        }
    }
    cout << "\nMaximum Utility Value: " << dp[N][W] << endl;

    return 0;
}



Dynamic programming
space & time = O(N x W) no. of items x weight capacity of truck
