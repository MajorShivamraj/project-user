#include <bits/stdc++.h>
using namespace std;

struct Order {
    string order_id;
    time_t timestamp;

    void display() const {
        cout << "Order ID: " << order_id << ", Timestamp: " << ctime(&timestamp);
    }
};

// Generate random orders
vector<Order> generate_orders(int n) {
    vector<Order> orders(n);
    srand(time(0));

    tm base_time = {};
    base_time.tm_year = 2025 - 1900; 
    base_time.tm_mon = 5;            // June 
    base_time.tm_mday = 24;
    base_time.tm_hour = 12;
    base_time.tm_min = 0;
    base_time.tm_sec = 0;

    time_t base = mktime(&base_time);

    for (int i = 0; i < n; i++) {
        orders[i].order_id = "ORD" + to_string(i + 1);
        // Randomize hours, minutes, seconds within ~70 days
        int random_seconds = rand() % (70 * 24 * 60 * 60); 
        orders[i].timestamp = base + random_seconds;
    }
    return orders;
}

void merge(vector<Order> &orders, int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;

    vector<Order> L(n1), R(n2);
    for (int i = 0; i < n1; i++) L[i] = orders[left + i];
    for (int i = 0; i < n2; i++) R[i] = orders[mid + 1 + i];

    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2) {
        if (L[i].timestamp <= R[j].timestamp) orders[k++] = L[i++];
        else orders[k++] = R[j++];
    }

    while (i < n1) orders[k++] = L[i++];
    while (j < n2) orders[k++] = R[j++];
}

// Merge sort function
void merge_sort(vector<Order> &orders, int left, int right) {
    if (left < right) {
        int mid = left + (right - left) / 2;
        merge_sort(orders, left, mid);
        merge_sort(orders, mid + 1, right);
        merge(orders, left, mid, right);
    }
}

int main() { 
    int NUM_ORDERS = 1000000;
    cout << "Generating orders...\n";
    vector<Order> orders = generate_orders(NUM_ORDERS);
    
    cout << "Sorting orders by timestamp...\n";
    clock_t start = clock();
    merge_sort(orders, 0, orders.size() - 1);
    clock_t end = clock();
    double time_taken = double(end - start) / CLOCKS_PER_SEC;
    cout << "Sorted " << NUM_ORDERS << " orders in " << time_taken << " seconds.\n";
    cout << "\nAll sorted orders:\n";
    
    for (int i=0;i<=30;++i) orders[i].display();
    return 0;
    }
